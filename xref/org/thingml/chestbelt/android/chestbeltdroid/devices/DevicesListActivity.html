271 	private final BroadcastReceiver btManagerReceiver = new BroadcastReceiver() {
272 		public void onReceive(Context context, Intent intent) {
273 			String action = intent.getAction();
274 			if (BluetoothManagementService.ACTION_DISCOVERY_STARTED.equals(action)) {
275 				Log.d(TAG, "Receiver: ACTION_DISCOVERY_STARTED");
276 				showDialog(DIALOG_DISCOVERY_ID);
277 			} else if (BluetoothManagementService.ACTION_DISCOVERY_ENDED.equals(action)) {
278 				Log.d(TAG, "Receiver: ACTION_DISCOVERY_ENDED");
279 				dismissDialog(DIALOG_DISCOVERY_ID);
280 			} else if (BluetoothManagementService.ACTION_DEVICE_DETECTED.equals(action)) {
281 				Log.d(TAG, "Receiver: ACTION_DEVICE_DETECTED");
282 				deviceDetected(intent);
283 			} else if (BluetoothManagementService.ACTION_CONNECTION_SUCCESS.equals(action)) {
284 				Log.d(TAG, "Receiver: ACTION_CONNECTION_SUCCESS");
285 				dismissDialog(DIALOG_CONNECTION_ID);
286 				String address = intent.getExtras().getString(Device.EXTRA_DEVICE_ADDRESS);
287 				Device d = Device.getFromAddress(devices, address);
288 				deviceConnected(d);
289 			} else if (action.equals(BluetoothManagementService.ACTION_CONNECTION_FAILURE)) {
290 				Log.d(TAG, "Receiver: ACTION_CONNECTION_FAILURE");
291 				dismissDialog(DIALOG_CONNECTION_ID);
292 			} else if (BluetoothManagementService.ACTION_DEVICE_DISCONNECTED.equals(action)) {
293 				Log.d(TAG, "Receiver: ACTION_DEVICE_DISCONNECTED");
294 				String address = intent.getExtras().getString(Device.EXTRA_DEVICE_ADDRESS);
295 				Device d = Device.getFromAddress(devices, address);
296 				deviceDisconnected(d);
297 			}
298 		}
299 	};

301 	private void deviceDetected(Intent intent) {
302 		String name = intent.getExtras().getString(Device.EXTRA_DEVICE_NAME);
303 		String address = intent.getExtras().getString(Device.EXTRA_DEVICE_ADDRESS);
304 		if (!Device.contain(devices, address)) { 
305 			Device device = new Device(name, address);
306 			Log.d(TAG, "New device added: " + name + " (" + address + ")");
307 			devices.add(device);
308 		} 
309 		Device d = Device.getFromAddress(devices, address);
310 		d.setConnected(intent.getExtras().getBoolean(BluetoothManagementService.EXTRA_DEVICE_IS_CONNECTED, false));
311 		d.setAvailable(d.isConnected() || intent.getExtras().getBoolean(BluetoothManagementService.EXTRA_DEVICE_IS_AVAILABLE));
312 		Log.i(TAG, "Device: " + d);
313 		deviceAdapter.notifyDataSetChanged();
314 	}

316 	private void deviceConnected(Device device) {
317 		Log.i(TAG, "Device connected: " + device.getName() + " (" + device.getAddress() + ")");
318 		device.setConnected(true);
319 		device.setAvailable(true);
320 		deviceAdapter.notifyDataSetChanged();
321 		if (changeActivity) {
322 			changeActivity = false;
323 			Intent i = new Intent(getApplicationContext(), DashBoardActivity.class);
324 			i.putExtra(Device.EXTRA_DEVICE_NAME, device.getName());
325 			i.putExtra(Device.EXTRA_DEVICE_ADDRESS, device.getAddress());
326 			startActivity(i);
327 		}
328 	}
